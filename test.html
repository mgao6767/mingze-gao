import numpy as np
import pandas as pd
from plotnine import *

np.random.seed(42)

# Simulation parameters
n_units_per_group = 5
n_periods = 6  # -3,...,+2
beta = 3       # true treatment effect
trend_diff = 0.5  # treatment group grows faster pre-shock

# Construct panel
units = list(range(n_units_per_group*2))
groups = ['Control']*n_units_per_group + ['Treatment']*n_units_per_group
time = np.arange(-3, 3)  # relative time

data = []
for i, g in zip(units, groups):
    for t in time:
        base = 10
        # Pre-trend difference
        pre_trend = trend_diff * max(t, 0) if g=='Treatment' else 0
        y = base + pre_trend + np.random.normal(0, 1)
        # Add post-shock effect
        y += beta if g=='Treatment' and t>=0 else 0
        data.append({'unit':i, 'group':g, 'time':t, 'y':y})

df = pd.DataFrame(data)

# Placebo test: pretend shock happened at t=-2
df['placebo'] = ((df['time']>=-2) & (df['group']=='Treatment')).astype(int)

# Run placebo regression
import statsmodels.formula.api as smf
placebo_model = smf.ols('y ~ placebo + C(group) + C(time)', data=df).fit()
print(placebo_model.summary())

import numpy as np
import pandas as pd
from plotnine import *

np.random.seed(42)

# Parameters
n_units = 20
n_periods = 6
treated_units = 10
beta = 3  # True ATT

# Simulate panel data
units = np.arange(n_units)
time = np.arange(-3, 3)  # relative time
df = pd.DataFrame([(i, t) for i in units for t in time], columns=['unit','time'])
df['group'] = np.where(df['unit'] < treated_units, 'Treatment','Control')

# Outcomes
# Systematically higher outcome for treated group
df['y'] = 10 + 0.2*df['time'] + (df['group']=='Treatment')*5 + (df['group']=='Treatment')*beta*(df['time']>=0) \
          + np.random.normal(0,1,len(df))

# Actual ATT
treated = df[df['group']=='Treatment']
control = df[df['group']=='Control']
tau_true = (treated[treated['time']>=0]['y'].mean() - treated[treated['time']<0]['y'].mean()) - \
           (control[control['time']>=0]['y'].mean() - control[control['time']<0]['y'].mean())
print("True ATT:", tau_true)

# Placebo simulation
placebo_estimates = []
for _ in range(1000):
    rand_treated = np.random.choice(units, size=treated_units, replace=False)
    df['placebo'] = df['unit'].isin(rand_treated)
    treated_r = df[df['placebo']]
    control_r = df[~df['placebo']]
    tau_r = (treated_r[treated_r['time']>=0]['y'].mean() - treated_r[treated_r['time']<0]['y'].mean()) - \
            (control_r[control_r['time']>=0]['y'].mean() - control_r[control_r['time']<0]['y'].mean())
    placebo_estimates.append(tau_r)

# Create DataFrame for plotting
placebo_df = pd.DataFrame({'DiD': placebo_estimates})

# Plot
(
    ggplot(placebo_df, aes(x='DiD')) +
    geom_histogram(binwidth=0.5, fill='#80225F', color='#373A36', alpha=0.7) +
    geom_vline(xintercept=tau_true, color='#A6192E', linetype='dashed', size=1.2) +
    labs(title='Randomized Placebo Test Distribution',

    )
)

import numpy as np
import pandas as pd
from plotnine import *

np.random.seed(42)

# Parameters
n_units = 20
n_periods = 6
treated_units = 10
beta = 3  # True ATT
pre_trend = 2  # slope in pre-period for treated

# Simulate panel data
units = np.arange(n_units)
time = np.arange(-3, 3)  # relative time
df = pd.DataFrame([(i, t) for i in units for t in time], columns=['unit','time'])
df['group'] = np.where(df['unit'] < treated_units, 'Treatment','Control')

# Outcomes with pretrend
df['y'] = 10 + 0.2*df['time'] \
          + (df['group']=='Treatment')*(pre_trend*np.minimum(df['time'],0) + 5 + beta*(df['time']>=0)) \
          + np.random.normal(0,1,len(df))

# Compute average and 95% CI per group per time
summary = df.groupby(['group','time']).agg(
    mean_y=('y','mean'),
    std_y=('y','std'),
    n=('y','count')
).reset_index()
summary['se'] = summary['std_y']/np.sqrt(summary['n'])
summary['ymin'] = summary['mean_y'] - 1.96*summary['se']
summary['ymax'] = summary['mean_y'] + 1.96*summary['se']

# Actual ATT (for reference)
treated = df[df['group']=='Treatment']
control = df[df['group']=='Control']
tau_true = (treated[treated['time']>=0]['y'].mean() - treated[treated['time']<0]['y'].mean()) - \
           (control[control['time']>=0]['y'].mean() - control[control['time']<0]['y'].mean())

# Placebo simulation (randomized groups)
placebo_estimates = []
for _ in range(1000):
    rand_treated = np.random.choice(units, size=treated_units, replace=False)
    df['placebo'] = df['unit'].isin(rand_treated)
    treated_r = df[df['placebo']]
    control_r = df[~df['placebo']]
    tau_r = (treated_r[treated_r['time']>=0]['y'].mean() - treated_r[treated_r['time']<0]['y'].mean()) - \
            (control_r[control_r['time']>=0]['y'].mean() - control_r[control_r['time']<0]['y'].mean())
    placebo_estimates.append(tau_r)
placebo_df = pd.DataFrame({'DiD': placebo_estimates})

# Top panel: average outcomes with CI
p1 = (
    ggplot(summary, aes(x='time', y='mean_y', color='group', group='group')) +
    geom_line(size=1.5) +
    geom_ribbon(aes(ymin='ymin', ymax='ymax', fill='group'), alpha=0.2, color=None) +
    geom_vline(xintercept=0, linetype='dashed', color='#373A36', size=1) +
    scale_color_manual(values=['#A6192E','#80225F']) +
    scale_fill_manual(values=['#A6192E','#80225F']) +
    scale_x_continuous(breaks=np.arange(-3,4,1)) +
    labs(title='Treated vs Control: Pre-trend Violation',
         subtitle=f'ATT = {tau_true:.2f}, pre-trend slope={pre_trend}',
         x='Time relative to shock',
         y='Average outcome') +
    theme_minimal() +
    theme(panel_grid_major=element_blank(),
          panel_grid_minor=element_blank(),
          axis_text=element_text(size=10),
          axis_title=element_text(size=12),
          plot_title=element_text(size=14, weight='bold'),
          plot_subtitle=element_text(size=12),
          legend_position='right')
)

# Bottom panel: placebo histogram
p2 = (
    ggplot(placebo_df, aes(x='DiD')) +
    geom_histogram(binwidth=0.5, fill='#80225F', color='#373A36', alpha=0.7) +
    geom_vline(xintercept=tau_true, color='#A6192E', linetype='dashed', size=1.2) +
    labs(title='Randomized Placebo Test Distribution',
         x='DiD Estimate (Placebo Groups)',
         y='Frequency') +
    theme_minimal() +
    theme(panel_grid_major=element_blank(),
          panel_grid_minor=element_blank())
)
